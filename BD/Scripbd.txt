CREATE TABLE TIPO_USUARIO(
 ID INT AUTO_INCREMENT PRIMARY KEY NOT NULL, 
DOCENTE VARCHAR(50) NOT NULL,
ESTUDIANTE VARCHAR(100) NOT NULL,
ADMINISTRADOR VARCHAR(500) NOT NULL
);



CREATE TABLE USUARIO(
ID INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
 NOMBRE VARCHAR(200) NOT NULL,
 CORREO VARCHAR(200) NOT NULL,
 CONTRASEÃ‘A VARCHAR(200) NOT NULL,
 TIPO_USUARIO VARCHAR(200) NOT NULL,    
FOREIGN KEY (TIPO_USUARIO) REFERENCES TIPO_USUARIO(ID)
);


CREATE TABLE ID_TEMA(
ID_TEMA  INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
 TEORIA1 VARCHAR(200) NOT NULL,
 TEORIA2 VARCHAR(200) NOT NULL,
 TEORIA3 VARCHAR(200) NOT NULL,
FOREIGN KEY (ID_TEMA) REFERENCES USUARIO(ID)
);





CREATE TABLE QUIZ (
ID_EJERCICIO INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
 NIVEL1 VARCHAR(200) NOT NULL,
 NIVEL2 VARCHAR(200) NOT NULL,
 NIVEL3 VARCHAR(200) NOT NULL,
FOREIGN KEY (ID_EJERCICIO) REFERENCES USUARIO(ID)
);


CREATE TABLE  NIVEL (
ID_NIVEL INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
 PREGUNTAS  VARCHAR(200) NOT NULL,
FOREIGN KEY (ID_NIVEL) REFERENCES  QUIZ(ID_EJERCICIO)
);
